  } else {

  }

export const markdownLinkExtractor = (markdown, route) => {
  let links = [];
  let renderer = new marked.Renderer();
  renderer.link = function (href, title, text) {
    links.push({
      href: href,
      path: route,
      text: text.substr(0,50)});
  };
  marked(markdown, { renderer: renderer });
  return links
};
